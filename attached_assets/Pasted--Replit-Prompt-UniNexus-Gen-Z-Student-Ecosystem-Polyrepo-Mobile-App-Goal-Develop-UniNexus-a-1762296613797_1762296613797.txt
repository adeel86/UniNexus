ðŸ§° Development Setup (Replit Command Template)
# 1. Initialise Polyrepo
mkdir uninexus && cd uninexus

# Create sub-repositories
npx create-expo-app uninexus-frontend
mkdir uninexus-backend uninexus-ai-services uninexus-auth-service uninexus-db-schema uninexus-devops uninexus-admin-panel

# 2. Initialise Git for each service
for dir in uninexus-*; do
  cd $dir
  git init
  echo "node_modules" > .gitignore
  cd ..
done

# 3. Link Remote Repos (GitHub)
# Example: git remote add origin https://github.com/username/uninexus-frontend.git

# 4. Setup Docker and CI/CD
cd uninexus-devops
touch docker-compose.yml
touch .github/workflows/deploy.yml

# 5. Launch Local Dev
cd ../uninexus-frontend
npx expo start
